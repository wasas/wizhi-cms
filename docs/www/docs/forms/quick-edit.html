    $fm = new Fieldmanager_TextField( array(
        'name' => 'demo',
    ) );
    $fm->add_quickedit_box( 'Basic Field', 'post', function( $values ) {
        return ! empty( $values['demo'] ) ? $values['demo'] : '';
    } );
} );
